\section{Debugging tricks}
	\begin{itemize}
		\item \mintinline{c++}{signal(SIGSEGV, [](int) { _Exit(0); });} converts segfaults into Wrong Answers.
			Similarly one can catch SIGABRT (assertion failures) and SIGFPE (zero divisions).
			\mintinline{c++}{_GLIBCXX_DEBUG} failures generate SIGABRT (or SIGSEGV on gcc 5.4.0 apparently).
		\item \mintinline{c++}{feenableexcept(29);} kills the program on NaNs (\texttt 1), 0-divs (\texttt 4), infinities (\texttt 8) and denormals (\texttt{16}).
	\end{itemize}

\section{Optimization tricks}
	\mintinline{c++}{__builtin_ia32_ldmxcsr(40896);} disables denormals (which make floats 20x slower near their minimum value).
	
	\subsection{Bit hacks}
		\begin{itemize}
			\item \mintinline{c++}{x & -x} is the least bit in \texttt{x}.
			\item \mintinline{c++}{for (int x = m; x; ) { --x &= m; ... }} loops over all subset masks of \texttt{m} (except \texttt{m} itself).
			\item \mintinline{c++}{c = x&-x, r = x+c; (((r^x) >> 2)/c) | r} is the next number after \texttt{x} with the same number of bits set.
			\item \mintinline{c++}{rep(b,0,K) rep(i,0,(1 << K))} \\ \mintinline{c++}{  if (i & 1 << b) D[i] += D[i^(1 << b)];} computes all sums of subsets.
		\end{itemize}
	
	\subsection{Pragmas}
		\begin{itemize}
			\item \mintinline{c++}{#pragma GCC optimize ("Ofast")} will make GCC auto-vectorize loops and optimizes floating points better.
			\item \mintinline{c++}{#pragma GCC target ("avx2")} can double performance of vectorized code, but causes crashes on old machines.
			\item \mintinline{c++}{#pragma GCC optimize ("trapv")} kills the program on integer overflows (but is really slow).
		\end{itemize}
